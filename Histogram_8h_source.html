35  Histogram(double Emin, double dE, double Emax)
36  : Emin(Emin), dE(dE), dEinv(1. / dE), nE(int(ceil((Emax - Emin) / dE))), out(nE, 0.)
37  {}
38 
39  //Alternate two-stage addEvent logic which is useful for arrays of similar histograms
40  bool eventPrecalc(double E, int& iEvent, double& tEvent)
41  {
42  double eCenter = (E - Emin)*dEinv;
43  iEvent = floor(eCenter);
44  if (iEvent<0 || iEvent + 1 >= nE) return false;
45  tEvent = eCenter - iEvent;
46  return true;
47  }
48  inline void addEventPrecalc(int iEvent, double tEvent, double weight)
49  {
50  double prefac = dEinv * weight;
51  out[iEvent] += prefac * (1. - tEvent);
52  out[iEvent + 1] += prefac * tEvent;
53  }
54 
55  void addEvent(double E, double weight)
56  { //Linear splined histogram
57  //--- E coordinate:
58  int iEvent; double tEvent;
59  if (not eventPrecalc(E, iEvent, tEvent)) return;
60  //--- accumulate normalized linear spline:
61  addEventPrecalc(iEvent, tEvent, weight);
62  }

64  void print(string fname, double Escale, double histScale) const
65  {
66  FILE* fp = fopen(fname.c_str(), "w");
67  for (size_t i = 0; i < out.size(); i++)
68  fprintf(fp, "%14.7le %14.7le\n", (Emin + i*dE)*Escale, out[i] * histScale);
69  fclose(fp);
70  }
71 };
