147  fprintf(fp, what);

149  for (int i = 0; i < dimension; i++)
150  for (int j = 0; j < dimension; j++){
151  if (i == j && abs(m[i][i]) < diag_min) { diag_min = abs(m[i][i]); }
152  if (i != j && abs(m[i][j]) > off_max) { off_max = abs(m[i][j]); }
153  }
154  if (drop_tiny_off && off_max < thr * diag_min){
